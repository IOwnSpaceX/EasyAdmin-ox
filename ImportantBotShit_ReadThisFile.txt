==============================
  Example - giving a role all ban permissions:
REPLACE "ROLEID" with your role ids. (NO QUOTES!)
==============================

add_ace role:ROLEID easyadmin.bot.ban allow
add_ace role:ROLEID easyadmin.player.ban.temporary allow
add_ace role:ROLEID easyadmin.player.ban.permanent allow
add_ace role:ROLEID easyadmin.bot.unban allow
add_ace role:ROLEID easyadmin.player.ban.remove allow
add_ace role:ROLEID easyadmin.bot.baninfo allow
add_ace role:ROLEID easyadmin.player.ban.view allow

==============================
  Required in server.cfg for add/remove_principal to work:
==============================

add_ace resource.EasyAdmin-ox command.add_principal allow
add_ace resource.EasyAdmin-ox command.remove_principal allow
